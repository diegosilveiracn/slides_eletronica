\begin{frame}
	\frametitle{Eletricidade}
	
	\begin{itemize}
		\item É uma forma de energia resultante do movimento de partículas carregadas;
		\item Essa energia pode ser gerada por fontes naturais ou artificiais;
		\item É essencial para o funcionamento de dispositivos e sistemas no mundo moderno; e
		\item Pode ser classificada como:
		\begin{itemize}
			\item Estática; ou
			\item Dinâmica.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Materiais elétricos}
	
	\begin{itemize}
		\item Exitem diferentes tipos de materiais e suas propriedades fundamentais:
		\begin{itemize}
			\item Condutores;
			\item Isoladores; e
			\item Semicondutores.
		\end{itemize} 
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Condutores}
	
	\begin{itemize}
		\item São os materiais elétricos que permitem a passagem de corrente elétrica com facilidade. 
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplos}
		\begin{itemize}
			\item Cobre;
			\item Alumínio; ou
			\item Prata.
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}
	\frametitle{Isoladores}
	
	\begin{itemize}
		\item São materiais elétricos que não permitem a passagem de corrente elétrica com facilidade.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplos}
		\begin{itemize}
			\item Borracha;
			\item Plástico;
			\item Vidro; ou
			\item Cerâmica.
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}
	\frametitle{Semicondutores}
	
	\begin{itemize}
		\item Ocupam uma posição intermediária entre condutores e isolantes; e
		\item Apresenta uma capacidade limitada, mas controlável, de conduzir corrente elétrica.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplos}
		\begin{itemize}
			\item Germânio;
			\item Silício; ou
			\item Arseneto de gálio.
		\end{itemize}
	\end{exampleblock}
\end{frame}

\begin{frame}
	\frametitle{Tensão}
	
	\begin{itemize}
		\item É a força que impulsiona as cargas elétricas através de um condutor;
		\item Representa a diferença de energia elétrica entre dois pontos de um circuito;
		\item Para que a corrente elétrica flua, deve existir uma \structure{diferença de potencial} entre esses pontos, o que gera um movimento de elétrons; e
		\item A unidade de medida da tensão elétrica é o \structure{Volt $(V)$}.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Corrente}

	\begin{itemize}
		\item A corrente elétrica é o fluxo de cargas elétricas através de um material condutor;
		\item \alert{O sentido real da corrente parte do polo negativo para o polo positivo};
		\item A unidade de medida da corrente elétrica é o \structure{Ampère $(A)$}; e
		\item Outras medidas:
		\begin{itemize}
			\item Milliampère ($mA$) = $1/1000A$; e
			\item Microampère ($\mu A$) = $1/1.000.000A$
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Resistência}
	
	\begin{itemize}
		\item É a oposição ao fluxo de corrente elétrica dentro de um material;
		\item Ela é gerada pelas interações entre as partículas do material e as cargas elétricas, que dificultam o movimento das cargas;
		\item A unidade de medida da resistência elétrica é o \structure{Ohm $(\Omega)$}; e
		\item Outras medidas:
		\begin{itemize}
			\item Quilo-ohm($k\Omega$) = $1000\Omega$; e
			\item Mili-ohm ($\mu\Omega$)= $1/1000\Omega$.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Lei de Ohm}
	
	\begin{itemize}
		\item Estabelece a relação entre tensão elétrica (V), corrente elétrica (I) e resistência elétrica (R) em um circuito; e
		\item É a base para entender e calcular o comportamento de circuitos elétricos.
	\end{itemize}\vfill
	
	\begin{block}{Fórmula}
		$V = I \times R$
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Potência}
	
	\begin{itemize}
		\item É a medida da quantidade de energia elétrica transformada ou consumida por um dispositivo em um determinado intervalo de tempo;
		\item Indica a taxa na qual a energia é convertida em outras formas, como luz, calor ou movimento; e
		\item A unidade de medida da potência elétrica é o \structure{Watt (W)}.
	\end{itemize}\vfill
	
	\begin{block}{Fórmula}
		$P = V \times I$
	\end{block}
\end{frame}

\begin{frame}
	\frametitle{Primeira Lei de Kirchhoff}
	
	\begin{itemize}
		\item Também conhecida como Lei dos Nós;
		\item Estabelece que a soma das correntes elétricas que entram em um nó é igual à soma das correntes que saem dele; e
		\item Um nó é um ponto em um circuito onde três ou mais condutores se encontram.
	\end{itemize}\vfill
	
	\begin{columns}
		\column{0.35\textwidth}
		\begin{block}{Fórmula}
			$I_{1} + I_{2} + \dots + I_{n} = I_{3} + I_{4} + \dots + I_{m}$
		\end{block}
		
		\column{0.4\textwidth}
		\includegraphics[scale=0.35]{img/primeira_lei_Kirchhoff}
	\end{columns}
\end{frame}

\begin{frame}
	\frametitle{Segunda Lei de Kirchhoff}
	
	\begin{itemize}
		\item Também conhecida como Lei das Malhas;
		\item A firma que qualquer malha fechada de um circuito, a soma algébrica das tensões é igual a zero; e
		\item A quantidade de energia fornecida pelas fontes de tensão em um circuito é igual à quantidade de energia consumida pelos componentes.
	\end{itemize}\vfill
	
	\begin{columns}[c]
		\column{0.25\textwidth}
		\begin{block}{Fórmula}
			$V_{1} + V_{2} + \dots + V_{n} = 0$
		\end{block}
		
		\column{0.4\textwidth}
		\includegraphics[scale=0.35]{img/segunda_lei_Kirchhoff}
	\end{columns}
\end{frame}

\begin{frame}
	\frametitle{Capacitância}
	
	\begin{itemize}
		\item É a capacidade de um componente elétrico de armazenar energia elétrica na forma de um campo eletrostático;
		\item Quando um capacitor é ligado a uma fonte de tensão (V), ele começa a acumular carga elétrica;
		\item Se a fonte de tensão for removida, o capacitor funciona como uma especie de bateria temporária;
		\item A unidade de medida de um capacitor é o \structure{Farad (F)}; e
		\item Outra medidas:
		\begin{itemize}
			\item Microfarad $(\mu F) = 10^{-6} F$;
			\item Nanofarad $(nF) = 10^{-9} F$; e
			\item Picofarad $(pF) = 10^{-12} F$.
		\end{itemize}
	\end{itemize}
\end{frame}
